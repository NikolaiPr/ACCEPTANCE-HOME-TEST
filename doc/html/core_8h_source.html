149 textfile_format GetFileFormat(const std::string &file)
150 {
151 std::FILE* pFile = nullptr;
152 fopen_s(&pFile, file.c_str(), "r");
153 if (!pFile)
154 return textfile_format::error_file;
155
156 int c;
157 int format = 0;
158 int i = 0;
159
160 while ((c = std::fgetc(pFile)) != EOF && i < 4)
161 {
162 format |= (c << i * 8);
163 switch (format)
164 {
165 case 0xbfbbef: return textfile_format::utf8_file; //0xefbbbf
166 case 0x762f2b: return textfile_format::utf7_file; //0x2b2f76
167 case 0xfeff: return textfile_format::unicode_file; //0xfffe
168 case 0xfffe: return textfile_format::bigendianunicode_file;//0xfeff
169 case 0xfffe0000: return textfile_format::utf32_file; //0x0000feff
170 }
171 i++;
172 }
173
174 std::fclose(pFile);
175 return textfile_format::ascii_file;
176 }
