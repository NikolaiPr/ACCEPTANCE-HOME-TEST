139 textfile_format GetFileFormat(const std::string &file)
140 {
141 std::FILE* pFile = nullptr;
142 fopen_s(&pFile, file.c_str(), "r");
143 if (!pFile)
144 return textfile_format::error_file;
145
146 int c;
147 int format = 0;
148 int i = 0;
149
150 while ((c = std::fgetc(pFile)) != EOF && i < 4)
151 {
152 format |= (c << i * 8);
153 switch (format)
154 {
155 case 0xbfbbef: return textfile_format::utf8_file; //0xefbbbf
156 case 0x762f2b: return textfile_format::utf7_file; //0x2b2f76
157 case 0xfeff: return textfile_format::unicode_file; //0xfffe
158 case 0xfffe: return textfile_format::bigendianunicode_file;//0xfeff
159 case 0xfffe0000: return textfile_format::utf32_file; //0x0000feff
160 }
161 i++;
162 }
163
164 std::fclose(pFile);
165 return textfile_format::ascii_file;
166 }
